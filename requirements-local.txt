# Optional dependencies for the free, on-device backends.
# Install with:  pip install -r requirements-local.txt
#
# These power VOICE_BACKEND=local (faster-whisper → Ollama → Kokoro) and the
# voice side of the local stack. The LLM brain runs in a separate Ollama
# server — install it from https://ollama.com and pull your models, e.g.:
#
#     ollama pull qwen3:8b          # the chat / agentic model (LOCAL_LLM_MODEL)
#     ollama pull qwen2.5vl:7b      # vision model for the skill compiler (optional)
#
# kokoro-onnx bundles espeak-ng (via espeakng-loader), so no system install is
# needed, and it supports Python 3.13. The ~340 MB Kokoro model files are
# downloaded automatically on first use to LOCAL_TTS_MODEL_PATH / _VOICES_PATH.
#
# Mirrors the `local` extra in pyproject.toml.

faster-whisper>=1.0      # speech-to-text (CTranslate2; runs on CPU on Apple Silicon)
kokoro-onnx>=0.4         # text-to-speech (Kokoro 82M, 24 kHz, ONNX runtime + bundled espeak)
webrtcvad-wheels>=2.0    # voice-activity / turn detection (prebuilt wheels)
soundfile>=0.12          # audio I/O
